@Override
public Map<Long, Book> findAllBook() throws Exception {
SqlSession session = MybatisSessionFactory.getSession();
BookMapper bm =
session.getMapper(BookMapper.class);
List<Book> books = bm.selectBooks();
Map<Long, Book> map = new TreeMap<>();
for(Book b:books){
map.put(b.getId(), b);
}
return map;
}
