public class SingleAppUsingJoobyTest {
static OkHttpClient client = new OkHttpClient();
public String serverPath;
public int serverPort;
@Test
public void sayHi() throws IOException {
Request request = new Request.Builder()
.url(serverPath)
.build();

}
}
